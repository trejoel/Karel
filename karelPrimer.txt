   void tomarZumbadores(){
      while(nextToABeeper){
         pickbeeper();
      }
   
   }
   
   void dobla(){
     turnleft();
     turnleft();
   }
   
   void turnright(){
     iterate(3){
       turnleft();
     }
   }
   
   void regresa_una_posicion(){
      dobla();
      move();
      turnright();
      move();
   }
   
   void avanzaHaciaCasa(){
     
     if (anyBeepersInBeeperBag){
         putbeeper(); 
         avanzaHaciaCasa();
         move();
         turnright();
         move();
         turnleft();
     }     
   }
