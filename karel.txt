    void putKarelNorth(){
      while (notFacingNorth){
         turnleft();
      }
    }

    void putKarelEast(){
      while (notFacingEast){
         turnleft();
      }
    }

    
     void putKarelWest(){

          turnleft();
       }
    }
    
    void putKarelSouth(){

    }
    
    void avanza(n){
      iterate(n){
        putbeeper();
        move();
      }
      putKarelWest();
      while(frontIsClear){
        move();
      }
      putKarelNorth();
      move();
    }
    
    void piramide(n){
        putKarelEast();
        avanza(n);
    }
    
        
    void altura(n){
      if (nextToABeeper){
